383      public static double hii(@NonNull String cur11, String cur22, @NonNull HashMap<Integer, CurrencyDetails> hashMap) {
384  
385          double h = 1;
386          String cur1 = "", cur2 = "";
387  
388          for (CurrencyDetails currencyDetails : hashMap.values()) {
389              if (currencyDetails.getName().equals(cur11)) {
390                  cur1 = currencyDetails.getCode();
391              }
392              if (currencyDetails.getName().equals(cur22)) {
393                  cur2 = currencyDetails.getCode();
394              }
395          }
396  
397  
398  
399              if (initSymbol().get(cur1).equals("*") && initSymbol().get(cur2).equals("*")) {
400  
401                  if (initCutPrice().get(cur1) > initCutPrice().get(cur2)) {
402                      h = initCutPrice().get(cur1) / initCutPrice().get(cur2);
403                  } else {
404                      h = initCutPrice().get(cur2) / initCutPrice().get(cur1);
405                  }
406              } else if (initSymbol().get(cur1).equals("/") && initSymbol().get(cur2).equals("/")) {
407                  if (initCutPrice().get(cur1) > initCutPrice().get(cur2)) {
408                      h = initCutPrice().get(cur2) / initCutPrice().get(cur1);
409                  } else {
410                      h = initCutPrice().get(cur1) / initCutPrice().get(cur2);
411                  }
412  
413                  h = initCutPrice().get(cur1) / initCutPrice().get(cur2);
414              } else {
415                  h = initCutPrice().get(cur1) * initCutPrice().get(cur2);
416              }
417  
418          return h;
419      }
