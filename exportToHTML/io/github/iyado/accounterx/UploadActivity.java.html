148      public void uploadData(@NonNull Activity activity,@NonNull String id,
149                                    @NonNull String username,
150                                    @NonNull String fullname,
151                                    double account ,
152                                    @NonNull String cur){
153  
154          SingleCurrency singleCurrency = new SingleCurrency(hello(account));
155          //singleCurrency.setCurrencyName(cur);
156  
157          Informations userinfo = new Informations(id, type(), fullname);
158  
159          long yourmilliseconds = System.currentTimeMillis();
160          @SuppressLint("SimpleDateFormat") SimpleDateFormat sdf = new SimpleDateFormat("MMM dd,yyyy");
161          Date resultdate = new Date(yourmilliseconds);
162          String time = String.valueOf(yourmilliseconds);
163          String date = sdf.format(resultdate);
164  
165          Prog progF = new Prog( username,username,account,
166                  cur,
167                  0,0,"افتتاحي","",
168                  (hello((account))),
169                  time,date);
170  
171          Prog progT = new Prog( username,username,account,
172                  cur,
173                  0,0,"افتتاحي","",
174                  -hello((account)),
175                  time,date);
176  
177          usersReference.addListenerForSingleValueEvent(new ValueEventListener() {
178              @Override
179              public void onDataChange(@NonNull DataSnapshot snapshot) {
180                      Informations informations = snapshot.getValue(Informations.class);
181                      Log.e("TAG", "onDataChange: "+informations.getUsername());
182                     // if (informations.getUid().equals(id)){
183                     //     isFound = true;
184               //   }
185              }
186  
187              @Override
188              public void onCancelled(@NonNull DatabaseError error) {
189  
190                  noti(UploadActivity.this,error.getMessage());
191              }
192          });
193  
194          usersReference.child(username)
195                          .get().addOnCompleteListener(task -> {
196                              if (task.getResult().exists()){
197                                  noti(UploadActivity.this,username+"\n اسم المستخدم موجود مسبقاَ");
198                              }else {
199                                  if (isFound) {
200                                      noti(UploadActivity.this, id + "\n رقم الصندوق موجود مسبقاَ");
201  
202                                  } else {
203                                      usersReference
204                                              .child(username)
205                                              .setValue(userinfo).addOnCompleteListener(taskx -> {
206                                                  if (taskx.isSuccessful()) {
207                                                      usersReference
208                                                              .child(username)
209                                                              .child("accounts")
210                                                              .child(date)
211                                                              .child(time)
212                                                              .setValue(progF)
213                                                              .addOnCompleteListener(runnable -> {
214                                                                  if (runnable.isSuccessful()) {
215                                                                      usersReference
216                                                                              .child(username)
217                                                                              .child("account")
218                                                                              .child(cur)
219                                                                              .setValue(singleCurrency);
220                                                                  }
221                                                              });
222  
223                                                      usersReference
224                                                              .child("رأس المال")
225                                                              .child("accounts")
226                                                              .child(date)
227                                                              .child(time)
228                                                              .setValue(progT)
229                                                              .addOnCompleteListener(runnablex -> {
230                                                                  if (runnablex.isSuccessful()) {
231                                                                      noti(activity.getApplicationContext(), " \n تمت اضافة" + username + " بنجاح");
232                                                                      activity.finish();
233                                                                  }
234                                                              });
235                                                  }else {
236                                                      if (taskx.getException()!= null) {
237                                                          noti(activity, taskx.getException().getMessage());
238                                                      }
239                                                  }
240                                              })
241                                              .addOnFailureListener(e ->noti(activity, e.getMessage()));
242                                  }
243                              }
244                          });
245      }
