65       @SuppressLint({"SetTextI18n", "SimpleDateFormat", "NotifyDataSetChanged"})
66       @Override
67       public void onBindViewHolder(@NonNull CViewHolder cViewHolder, @SuppressLint("RecyclerView") int position) {
68   
69           Prog prog = progs.get(position);
70   
71   
72   
73           cViewHolder.checkBox.setChecked(prog.isChecked());
74   
75   
76           /* 
77               ((PrintActivity)context).runOnUiThread(() -> { 
78                   cViewHolder.checkBox.setChecked(prog.isChecked()); 
79                 //  notifyDataSetChanged(); 
80               }); 
81    
82            */
83   
84          cViewHolder.checkBox.setOnCheckedChangeListener((buttonView, isChecked) -> {
85              if (prog.isChecked()){
86                  editcheck(prog,false);
87              }else {
88                  editcheck(prog,true);
89              }
90              prog.setChecked(isChecked);
91   
92   
93          });
94   
95           double m = 0;
96           for (int j=0; j<=position; j++){
97               m += progs.get(j).getSumAll();
98           }
99           cViewHolder.total.setText(" "+m);
100          cViewHolder.total.setTextColor(context.getColor(R.color.m));
101          if (prog.getSumAll() > 0 ){
102              cViewHolder.ex.setTextColor(context.getColor(R.color.x500));
103          }else if (prog.getSumAll() < 0){
104              cViewHolder.ex.setTextColor(context.getColor(R.color.red));
105          }else {
106              cViewHolder.ex.setTextColor(context.getColor(R.color.orange));
107          }
108          if (prog.getSender().equals(prog.getReciever())){
109              cViewHolder.imageView.setVisibility(View.INVISIBLE);
110          }
111          databaseReference1 = usersReference
112                          .child(prog.getSender());
113  
114          databaseReference2 = usersReference
115                          .child(prog.getReciever());
116  
117          cViewHolder.count.setText(prog.getEx());
118          cViewHolder.ex.setText(stringo(prog.getSumAll()));
119  
120          String hx;
121          if (prog.getReciever().equals(user)) {
122              hx = prog.getSender();
123          }
124          else {
125              hx = prog.getReciever();
126          }
127              cViewHolder.notice.setText(prog.getCustomer() + "\n" + hx
128                      + "\n" +
129                      new SimpleDateFormat("dd MMM yyyy HH:mm").format(new Date(Long.parseLong(prog.getKey()))));
130  
131  
132          cViewHolder.imageView.setOnClickListener(v -> new AlertDialog.Builder(v.getContext())
133                          .setMessage(R.string.sure)
134                          .setNeutralButton("نعم",(dialog, which) ->
135                                  databaseReference1.child("accounts")
136                                          .child(prog.getDate())
137                                          .child(prog.getKey()).removeValue()
138                                          .addOnSuccessListener(unused -> databaseReference2
139                                                  .child("accounts")
140                                                  .child(prog.getDate())
141                                                  .child(prog.getKey())
142                                                  .removeValue()
143                                                  .addOnSuccessListener(unused1 -> {
144                                                      notifyDataSetChanged();
145                                                      noti(context,"تم حذف القيد ذو بنجاح");
146                                                      ((Activity)context).finish();
147                                                  }).addOnFailureListener(e -> noti(v.getContext(), e.getMessage())))
148                                          .addOnFailureListener(e -> noti(v.getContext(), e.getMessage())))
149                          .setNegativeButton("الغاء",(dialog, which) ->
150                                  dialog.dismiss())
151                          .setCancelable(true)
152                          .setTitle("حذف القيد المحدد")
153                          .setIcon(android.R.drawable.ic_menu_delete)
154                          .show()
155          );
156  
157      }
