165      private void jumpx(){
166          Intent intent = new Intent(this,JobActivity.class);
167              intent.putExtra("cname",firebaseAuth.getCurrentUser().getUid());
168  
169              intent.addFlags(Intent.FLAG_ACTIVITY_NEW_TASK).addFlags(Intent.FLAG_ACTIVITY_CLEAR_TASK);
170              DatabaseReference reference = FirebaseDatabase.getInstance().getReference();
171              reference.child(FirebaseAuth.getInstance().getCurrentUser().getUid()).child("users").get().addOnCompleteListener(task -> {
172                  if (task.isSuccessful()) {
173                      if (task.getResult().exists()) {
174                          startActivity(intent);
175                      }else {
176                          setDefaultBoxes(this,FirebaseAuth.getInstance().getCurrentUser().getUid());
177                          startActivity(intent);
178                      }
179  
180              }else {
181                      accounterException = new AccounterException(task.getException().getMessage(),task.getException());
182                      alert.setText(accounterException.getLocalizedMessage());
183                      noti(this,accounterException.getLocalizedMessage());
184                  }
185                  }).addOnFailureListener(e -> {
186                      accounterException = new AccounterException(e.getMessage(),e);
187                      alert.setText(accounterException.getLocalizedMessage());
188                  noti(this,accounterException.getLocalizedMessage());
189              });
190         // startActivity(intent);
191      }
