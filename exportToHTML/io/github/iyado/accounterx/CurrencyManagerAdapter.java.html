15   import androidx.appcompat.app.AlertDialog;

18   import com.github.dhaval2404.colorpicker.model.ColorShape;

23   import java.util.List;

26   public class CurrencyManagerAdapter extends RecyclerView.Adapter<CurrencyManagerViewHolder>{
27   
28       private static final String TAG = "CurrencyManagerAdapter";
29       EditText nameEditText,codeEditText,evaluationPriceEditText;
30       Spinner operatorSpinner;
31       Button saveButton,cancelButton;
32       Button colorPickerButton;
33   
34       private final Context mContext;
35       public final List<CurrencyDetails> mCurrencyDetails;
36       public DatabaseReference cursReference;
37   
38       public CurrencyManagerAdapter(Context mContext, List<CurrencyDetails> mCurrencyManager) {
39           this.mContext = mContext;
40           this.mCurrencyDetails = mCurrencyManager;
41           cursReference = FirebaseDatabase.getInstance().getReference()
42                   .child(FirebaseAuth.getInstance().getCurrentUser().getUid()).child("currencies");
43       }
44   
45       @NonNull
46       @Override
47       public CurrencyManagerViewHolder onCreateViewHolder(@NonNull ViewGroup parent, int viewType) {
48           View view = LayoutInflater.from(parent.getContext()).inflate(R.layout.currency_manager,parent,false);
49           return new CurrencyManagerViewHolder(view);
50       }
51   
52       @Override
53       public void onBindViewHolder(@NonNull CurrencyManagerViewHolder holder, int position) {
54   
55   
56   
57           CurrencyDetails currencyDetails = mCurrencyDetails.get(position);
58           holder.rqm.setText(String.valueOf(currencyDetails.getId()));
59           holder.asm.setText(currencyDetails.getName());
60           holder.rmz.setText(currencyDetails.getCode());
61           holder.moaaml.setText(currencyDetails.getSymbol());
62           holder.saar.setText(String.valueOf(currencyDetails.getRate()));
63   
64           holder.view.setOnClickListener(v -> {
65               AlertDialog.Builder builder = new AlertDialog.Builder(mContext);
66               builder.setTitle("تعديل العملة");
67               builder.setMessage("هل تريد تعديل العملة");
68               builder.setCancelable(false);
69               builder.setNeutralButton("حذف", (dialog, which) -> {
70                   cursReference.child(currencyDetails.getName()).removeValue();
71               });
72               builder.setView(R.layout.add_currency);
73               builder.setPositiveButton("تعديل", (dialog, which) -> {
74                  DatabaseReference cur =  cursReference.child(currencyDetails.getName());
75                           cur.child("name").setValue(nameEditText.getText().toString());
76                           cur.child("code").setValue(codeEditText.getText().toString());
77                           cur.child("rate").setValue(Double.parseDouble(evaluationPriceEditText.getText().toString()));
78                           cur.child("symbol").setValue(operatorSpinner.getSelectedItem().toString());
79                           cur.child("color").setValue(colorPickerButton.getText().toString());
80                           dialog.dismiss();
81   
82               });
83               builder.setNegativeButton("الغاء", (dialog, which) -> {
84                   dialog.dismiss();
85                   notifyDataSetChanged();
86   
87                   });
88   
89               AlertDialog dialog = builder.create();
90   
91               dialog.show();
92               saveButton = dialog.findViewById(R.id.saveButton);
93               cancelButton = dialog.findViewById(R.id.cancelButton);
94               saveButton.setVisibility(View.GONE);
95               cancelButton.setVisibility(View.GONE);
96               colorPickerButton = dialog.findViewById(R.id.colorPickerButton);
97               colorPickerButton.setTextColor(Color.parseColor(currencyDetails.getColor()));
98               colorPickerButton.setText(currencyDetails.getColor());
99               colorPickerButton.setOnClickListener(vx -> new ColorPickerDialog.Builder(mContext)
100                      .setTitle("اختر اللون")
101                      .setColorShape(ColorShape.CIRCLE)
102                      .setColorListener((color, hex) -> {
103                          colorPickerButton.setBackgroundColor(Color.parseColor(hex));
104                          colorPickerButton.setText(hex);
105                          currencyDetails.setColor(hex);
106                      }).setPositiveButton("حفظ").setNegativeButton("الغاء").setDefaultColor(R.color.primaryTextColor).show());
107  
108  
109              operatorSpinner = dialog.findViewById(R.id.operatorSpinner);
110              operatorSpinner.setPrompt("اختر العملية");
111              String[] operators = {"/", "*"};
112              operatorSpinner.setAdapter(new ArrayAdapter<>(mContext, android.R.layout.simple_spinner_item, operators));
113  
114              if(currencyDetails.getSymbol().equals("/")){
115                  operatorSpinner.setSelection(0);
116              }else{
117                  operatorSpinner.setSelection(1);
118              }
119  
120              nameEditText = dialog.findViewById(R.id.nameEditText);
121              codeEditText = dialog.findViewById(R.id.codeEditText);
122              evaluationPriceEditText = dialog.findViewById(R.id.evaluationPriceEditText);
123  
124              nameEditText.setText(currencyDetails.getName());
125              codeEditText.setText(String.valueOf(currencyDetails.getCode()));
126              evaluationPriceEditText.setText(String.valueOf(currencyDetails.getRate()));
127  
128          });
129  
130  
131      }
132  
133      @Override
134      public int getItemCount() {
135          return mCurrencyDetails.size();
136      }
137  }
