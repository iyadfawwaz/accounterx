63       @SuppressLint("SuspiciousIndentation")
64       @Override
65       public void onBindViewHolder(@NonNull CursViewHolder holderx, @SuppressLint("RecyclerView") int position) {
66   
67   
68           double[] counts = new double[currencyDetailsList.size()];
69           for (int i = 0; i < currencyDetailsList.size(); i++) {
70               counts[i] = 0.0d;
71           }
72   
73           String currency = currencyDetailsList.get(currencyDetailsList.get(position).getId()).getName();
74           holderx.curx.setText(currency);
75           //noti(context, currency);
76           if (currencyDetailsList.get(currencyDetailsList.get(position).getId()).getColor() != null) {
77   
78               holderx.curx.setTextColor(Color.parseColor(currencyDetailsList.get(currencyDetailsList.get(position).getId()).getColor()));
79           }
80           if (user != null)
81           usersReference.child(user).child("account")
82                           .child(currency)
83                                   .addListenerForSingleValueEvent(new ValueEventListener() {
84                                       @SuppressLint("NotifyDataSetChanged")
85                                       @Override
86                                       public void onDataChange(@NonNull DataSnapshot snapshot) {
87   
88                                           if (!snapshot.exists()) {
89                                               counts[position] = 0.0d;
90                                               holderx.countx.setText(String.valueOf(counts[position]));
91                                               return;
92                                           }
93                                           SingleCurrency singleCurrency = snapshot.getValue(SingleCurrency.class);
94                                           singleCurrency.setCurrencyName(currency);
95                                           counts[position] = singleCurrency.getCount();
96                                           holderx.countx.setText(String.valueOf(counts[position]));
97                                           if (singleCurrency.getCount() < 0) {
98                                               holderx.countx.setTextColor(holderx.countx.getContext().getColor(R.color.red));
99                                           }
100                                          if (singleCurrency.getCount() == 0) {
101                                              holderx.countx.setTextColor(holderx.countx.getContext().getColor(R.color.orange));
102                                          }
103                                          if (singleCurrency.getCount() > 0) {
104                                              holderx.countx.setTextColor(holderx.countx.getContext().getColor(R.color.x400));
105                                          }
106                                          //notifyDataSetChanged();
107                                      }
108  
109                                      @SuppressLint("NotifyDataSetChanged")
110                                      @Override
111                                      public void onCancelled(@NonNull DatabaseError error) {
112  
113                                          holderx.countx.setText(error.getMessage());
114                                         // notifyDataSetChanged();
115                                      }
116                                  });
117
