53       private void saveCurrency() {
54           String name = nameEditText.getText().toString();
55           String code = codeEditText.getText().toString();
56           String operator = operatorSpinner.getSelectedItem().toString();
57           String evaluationPrice = evaluationPriceEditText.getText().toString();
58           String color = colorPickerButton.getText().toString();
59           curReference
60                   .addListenerForSingleValueEvent(new ValueEventListener() {
61                       @Override
62                       public void onDataChange(@NonNull DataSnapshot snapshot) {
63                           for(DataSnapshot dataSnapshot : snapshot.getChildren()){
64                               System.out.println(dataSnapshot.getValue(CurrencyDetails.class));
65                               CurrencyDetails currencyDetails = dataSnapshot.getValue(CurrencyDetails.class);
66                               System.out.println(currencyDetails.getName());
67                               currencyDetailslist.add(currencyDetails);
68   
69                           }
70   
71                           CurrencyDetails currencyDetails = new CurrencyDetails(currencyDetailslist.size(),name,code,operator,Double.parseDouble(evaluationPrice),"R.drawable.flag_egypt",color);
72                           curReference.child(name).setValue(currencyDetails);
73                           finish();
74   
75   
76                       }
77   
78                       @Override
79                       public void onCancelled(@NonNull DatabaseError error) {
80                           noti(AddCurrency.this,error.getMessage());
81   
82                       }
83                   });
