395      private void searchList(String newText) {
396          ArrayList<Prog> searchList = new ArrayList<>();
397          for (Prog prog: progME){
398              if (prog.getCustomer().toLowerCase().contains(newText.toLowerCase()) ||
399                      prog.getSender().toLowerCase().contains(newText.toLowerCase()) ||
400                      prog.getReciever().toLowerCase().contains(newText.toLowerCase()) ||
401                      String.valueOf(prog.getSumAll()).contains(newText.toLowerCase()) ||
402                      prog.getEx().toLowerCase().contains(newText.toLowerCase())){
403                  searchList.add(prog);
404              }
405          }
406          printerViewAdapter.setProgs(searchList);
407          printerViewAdapter.notifyDataSetChanged();
408      }

523      @SuppressLint("NotifyDataSetChanged")
524      private void loadUserInfo(String user, String cur) {
525          progs.clear();
526          progME.clear();
527          printerViewAdapter.setUser(user);
528          usersReference
529                  .child(user)
530                  .child("accounts")
531                  // .child("يوليو ١٧,٢٠٢٣")
532                  .addChildEventListener(new ChildEventListener() {
533                      @SuppressLint({"NotifyDataSetChanged", "SetTextI18n"})
534                      @Override
535                      public void onChildAdded(@NonNull DataSnapshot snapshot, @Nullable String previousChildName) {
536  
537                          // progME = new ArrayList<>();
538                          String s ;
539                          String t;
540                          double n;
541                          for (DataSnapshot dataSnapshot : snapshot.getChildren()) {
542                              progME.add(dataSnapshot.getValue(Prog.class));
543                              Prog prog = dataSnapshot.getValue(Prog.class);
544  
545                              assert prog != null;
546                              s = prog.getEx();
547                              t = prog.getDate();
548  
549                              Date date;
550                              try {
551                                  date = simpleDateFormat.parse(t);
552                              } catch (ParseException e) {
553                                  throw new RuntimeException(e);
554                              }
555  
556  
557                              if (s.equals(cur)) {
558                                  try {
559                                      assert date != null;
560                                      if (date.getTime() >= simpleDateFormat.parse(fx.getEditText().getText().toString()).getTime() && date.getTime() <= simpleDateFormat.parse(tx.getEditText().getText().toString()).getTime()) {
561                                          progs.add(prog);
562                                      }
563                                  } catch (ParseException e) {
564                                      throw new RuntimeException(e);
565                                  }
566                              }
567  
568                          }
569                          printerViewAdapter.notifyDataSetChanged();
570  
571  
572  
573  
574                      }
575  
576                      @SuppressLint("NotifyDataSetChanged")
577                      @Override
578                      public void onChildChanged(@NonNull DataSnapshot snapshot, @Nullable String previousChildName) {
579  
580                          printerViewAdapter.notifyDataSetChanged();
581                      }
582  
583                      @SuppressLint("NotifyDataSetChanged")
584                      @Override
585                      public void onChildRemoved(@NonNull DataSnapshot snapshot) {

588                      }
